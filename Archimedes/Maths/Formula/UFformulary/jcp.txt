Mandelbrotbis {
;
; Generic Mandelbrot set modified for z=1/z.
;
init:
  z = @start, c=0
loop:
  z = z^@power + #pixel*c
  c=@function(z)
  
bailout:
  |z| > @bailout
default:
  title = "Mandelbrotbis"
  center = (0,0)
  
  param start
   caption = "Starting point"
   default = (-1,0)
   endparam
  
  param power
    caption = "Power"
    default = (-1,0)
    endparam
 
 param function
 caption="Function"
 endparam
 
  float param bailout
    caption = "Bailout value"
    default =.01
    min = 0.1
    endparam
 switch:
  type = "Juliabis"
  seed = #pixel
  power = power
  function=function
  bailout = bailout
}
